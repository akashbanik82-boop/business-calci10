  <script>
    function calculateSimple() {
      let sales = Number(document.getElementById("sales").value);
      let expenses = Number(document.getElementById("expenses").value);
      let profit = sales - expenses;

      document.getElementById("simpleResult").innerText =
        "Monthly Profit: ₹" + profit;
    }

    function calculateAdvanced() {
      let raw = Number(document.getElementById("raw").value);
      let labour = Number(document.getElementById("labour").value);
      let electricity = Number(document.getElementById("electricity").value);
      let rent = Number(document.getElementById("rent").value);
      let emi = Number(document.getElementById("emi").value);
      let sales = Number(document.getElementById("advSales").value);

      let totalCost = raw + labour + electricity + rent + emi;
      let profit = sales - totalCost;

      document.getElementById("advancedResult").innerText =
        "Estimated Monthly Profit: ₹" + profit;
    }
  </script>
